    # .vimrc 常用配置

    " 语法高亮
syntax on
"自动对齐
set autoindent
"智能缩进
set smartindent

"tab缩进
set tabstop=4
set shiftwidth=4
set expandtab
set smarttab

" 256色
set t_Co=256

" 主题设置
colorscheme desert

" 插件 使用插件需要先安装git
call plug#begin('~/.vim/plugged')
Plug 'junegunn/vim-easy-align'
Plug 'ctrlpvim/ctrlp.vim'
call plug#end()
